 שוּ יַא ולַאד, לֵיש סַאכּתִין? לַא כַּלַאם עַלַא טַעַאם חַבּיבּתי קַצדֵכּ לַא סַלַאם עַלַא טַעַאם שוּ אַח'בַּאר? וַלַא אִשי, אַאה צַח אליוֹם וַזעוּ עַלֵינַא וַרַקַה קַאל רַח יִעמלוּ חַ לִלאַול ורַח יחֻּטוּ אִסמוֹ עַלַא לַאאִחת אלשרַף חַ וסַאכֵּת? מַבּרוּכּ מַבּרוּכּ סַנַד חַבּיבּי בַּאבַּא לִסא מַא טַלעוּ עַלַאמַאת אלאַולַ אלתַפַוק מַצ'מוּן חַבּיבּי קֻּל מִין אַשטַר מִנכּ בּאלצף אַאה? מִין? יַעני פִיה עִנדַכּ יוּסֵף שַאטֵר וּפיה סֵיף שַאטֵר ויַזַן כַּמַאן יַזַן? אַאה בַּאבַּא יַזַן ורַנַא כּתִיר שַאטרִין אַנַא בִּדי אַפהַם שוּ נַאקצַכּ אִנכּ תכּוּן אלאַול יַעני סַנַד בּשוּ אַחסַן מִנכּ אַאה? קֻּל בּשוּ אַחסַן מִנכּ? בַּאבַּא סַנַד שַאטֵר וּאַבּוּה כּתִיר בִּיסַאעדוֹ בּדִרַאסתוֹ אַבּוּה בִּיסַאעדוֹ כּתִיר אַקוּל לַכּ אַנַא עִנדי פִכּרַה אִבּעֵד הֵיכּ חַ אַעמַל סֵירתש סֵירתש עַלַא הַלק בַּאפַרגִ רַח תִטלַע אלאַול יַעני רַח תִטלַע אלאַול ואלחַ אִלנַא הַאלמַרַה מַאמַא מַעלֵיש אַטלַע עִנד יַזַן שוַי? לַא, לַאזֵם תֻּדרסֹ בַּאבַּא חַ דִרַאסתי רַאגֵ רַאגֵ מַא בִּדי חַדַא יתפַוק עַלֵיכּ בַּאבַּא, יַזַן אַבּוּה גַ 'אבּ לוֹ כּמֹבּיוּתר ג'דִיד קַאל רַח יסַאעדוֹ פִי אלדרַאסֵה בִּדי אַטלַע אַשוּפוֹ כִּיף? מִן וַין? שוּ יַעני בִּיעמַל אלכּמֹבּיוּתר? קַאל סַריע סַריע סַריע יַעני אַי סֻּאַאל בּתִעמַל לוֹ סֵירתש בִּיטַלע לַכּ אלאִגַ 'אבַּאת אִלי בּאלדניַא אַאה טַיבּ רוּח אֻּדרסֹ אַנַא בַּאשוּף לַכּ אלמַוצ'וּע עִצַאם? שוּ בּתִעמַל? וַלַא שִי בַּס בַּאתאַכּד אִזַא פִיה אִשי עַם בִּינַקט מִן פוֹק טַיבּ אַנַא טַאלעַה עִנד אִם יַזַן בִּדי אַגִ 'יבּ אלצאבּוּן אִלי וַציתהַא עַלֵיהַא חַ 'לַץ אַנַא בַּאטלַע אַג'יבּ לֵכּ אִיאה אַצלַן אַנַא בַּאחתַאג' מַפַכּ מִן עִנד אַבּו יַזַן אִרתַאחִי חַבּיבּתי בַּלַאש תתעבִּי חַאלֵכּ אַתעֵבּ חַאלי? אַהלֵין אַהלֵין אַבּוּ סַנַד כִּיפַכּ אַבּוּ יַזַן? כִּיף אלולַאד? שוּ עַם בִּידרַסוּ שִכּלהםֹ? יַענִי בִּיחַצ'רוּ חַאלהםֹ לִלאִמתִחַאנַאת חַרַאם עַלֵיכּ יַא זַלַמֵה מַא הֻּם טוּל אלנהַאר בּיֻּדרֻּסוּ בִּאלמַדרַסֵה כַּמַאן בִּאלבֵּית לַאזֵם יִלעַבּוּ יטַלעוּ טַאקִתהםֹ אַבּוּ יַזַן בִּדי אַסאַלַכּ, שוּ נוֹע אלכּמֹבּיוּתר אִלי גִ 'בּתוֹ ליַזַן צַדק אִני נַאסִי. בַּאשוּף לַכּ אִיאה, בּתִחתַאג' אִשי אַבּוּ סַנַד? אַאה צַאבּוּן אִם סַנַד אִזַא מַא פִיהַא עַ 'לַבֵּה קַאל נַאסי קַאל קַאל לַאזֵם יִלעַבּוּ. עַלַי? הַאי עֻּ 'רפֵת סַנַד מַצ'וִיה מקַטע חַאלוֹ דִרַאסֵה לַא וּבִּישַמשֵם עַן נוֹע אלכּמֹבּיוּתֵר אִסתעִד לִלהַזִימֵה יַא אַבּוּ סַנַד מַע מִין עַם תִחכּי? נַקזתִינִי, וַלַא מַע חַדַא מַע חַאלי בַּאחכִּי מַע חַאלי שוּ עַם בִּיצִיר אִסמַע יַאבַּא בִּתרַכּז בִּאלאִמתִחַאן מנִיח אִקרַא אלסאַאל מִית, אַקוּל לַכּ מִליוֹן מַרַה מַא בִּדי אַקַל מִן עַלַאמֵה כַּאמלֵה וּהַאדַא סַנַד אִלי בּיֻּדרסֹ מִן וַרַאכּ וּבִּיקוּל אַבּוּה אִנוֹ בִּילעַבּ בַּס הַדַפוֹ יִתפַוק עַלֵיכּ סִיבַּכּ מִנוֹ הַאדַא סַנַד חַבִּיבּי בִּדי אִיאכּ תפַלל עַלַאמֵה כַּאמלֵה אליוֹם והַאדַא צַאחבַּכּ יַזַן שוּ מפַכּר חַאלוֹ מֻּמכֵּן ינַאפסַכּ עַלַא חַאל אלכּמֹבּיוּתֵר תַבַּעַכּ אליוֹם בִּיוצַל ואַכִּיד רַח תִתפַוקַ עַלֵיה יַלא יַא ולַאד גַ 'הזוּ חַאלכּםֹ רַח אַוַזע וַרַק אלאִמתִחַאן טַלעוּ קַלַם רצַאץ ומַחאי ואלאַאלֵה אלחַאסבֵּה מִס נסִית אלאַאלֵה אלחַאסבֵּה תַבַּעתִי בִּאלבֵּית אַנַא נַבּהת עַלֵיכּםֹ מִית מַרַה מַא תִנסוּהַא. דַבּר חַאלַכּ מִס אַנַא גַ 'איֵבּ וַאחַדֵה זִיַאדֵה. סַנַד בִּיקדַר יִסתַח'דֵמהַא אַיוַה וִצֵל סַנַד אלעַבּקַרִי אִלי רַח יגִ 'יבּ לנַא אַחלַא חַ כִּיף יַאבַּא טַמננִי? בַּאבַּא לַו צַאחבּי מַא אַעטאני אַאלתוֹ אלחַאסבֵּה כַּאן סַקַטת בִּאלאִמתִחַאן אִנתֵ לַח'מתנִי אלצבּח ונסִית אַאלתִי אלחַאסבֵּה בִּאלבֵּית שוּ בִּתקוּל? כַּאן רַאח אלחַ אַללה סַתַר אִאִה מִין הַאדַא צַאחבַּכּ? לַאזֵם נִחכּי נֻּשכּרֹוֹ ונֻּשכּרֹ אַהלוֹ יַא סַלַאם הֵיכּ אלתרבַּאיֵה ולִא בַּלַאש הַאדַא צַאחבּי ורַח יצַ 'לוֹ צַאחבּי יַזַן ובַּאבַּא מַעלֵיש אִזַא אִנתֵ ועַמוֹ אַבּוּ יַזַן חַאבּין תִתנַאפַסוּ. תנַאפַסוּ בִּאלשע'ל ואִתרכּוּנַא נֻּדרסֹ בּרַאחתנַא חִלוּ כּתִיר אִנוֹ נשַג'ע ולַאדנַא יִבּדעוּ ויתפַוקוּ בַּס אלהַדַף אִנהםֹ יקַדמוּ אַחסַן אִשי עִנדהםֹ מֻּש אלהַדַף יכּוּנוּ אַחסַן מִן עֵ 'ירהםֹ אלמֻּקַארַנֵה ואלמֻּנַאפַסֵה אִזַא מַא כַּאנַת צַחיה מֻּש מנִיחַה לִולַאדנַא חַ 'לוּנַא נדִיר בַּאלנַא צַרַאחַה אַנַא מַצדוּמֵה מִן אִלי עִמלוּה זוַאג'נַא, אלעַ מִנהםֹ וִלא יַזַן מֻּש מַעקוּל שוּ סַריע הַאדַא אלכּמֹבּיוּתר סַריע כּתִיר בַּאקוּל לַכּ מַאן מֻּפִיד כּתִיר לִלדרַאסֵה כַּמַאן דִרַאסֵה קֻּלתוּ לִי, אַאה?
