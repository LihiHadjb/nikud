 סַאַלנַא עֵילת אַבּו סַנַד שוּ תעַלמתוּ פִי סִנת ושוּ כַאנַת אַחלַא לַחַטַ תעַלמת הַאי אלסנֵה אִנוֹ מַא אַקַארֵן חַאלִי בּאלנאס והַאדַא אלרגִ אַבּוּ אלתלַאת אַיאם מַא בִּינפַע מִן חַלקת: שִכלִי מֻש עַאגִ 'בּנִי אַלוֹ חַנִין, אִסמַעִי שוּ הַאדַא אלדִאֵית אִלי קֻלתִי לִי עַנוֹ אִלי נזִלתִי עַלֵיה כִילוֹ בִּתַלַאת אַיאם? עִידוּ וַרַאי, שִכלִי חִלוּ מִתֵל מַא אַנַא שִכלִי חִלוּ מִתֵל מַא אַנַא יַגִ 'בּ אַן תכוּן חַיַאתי צַחיה יַגִ 'בּ אַן תכוּן חַיַאתי צַחיה הַאי אִם זֵיד אִסם אַללה עַלֵיהַא שוּ נַחפַאנֵה פִכרכםֹּ שוּ אלדאֵית אִלי עַאמלֵתוֹ? בַּעדֵין אסתַנוּ שוַי, לֵיש בִּדכםֹּ אִיאנִי אַחמֵ אַנַא מֻש פַאהמֵה יַענִי עִנדִי אַרבַּע ולַאד, סַאלי וסַנַד וסַאמֵר ואַבּוּהםֹּ כַמַאן בַּעדֵין מפַכרִין אַבּוּ סַנַד מַא בִּיקרַא תַעלִיקַאתכםֹּ? הַאיוֹ צַאר ינֻק בִּדוֹ כַמַאן. עַאגֵ 'בּכםֹּ? מִן חַלקת: יוֹמכ חִלוּ אַו לַא? אַתטַלע עַלַא תַוקַעַאת אַבּרַאג' ולִא מַא אַתטַלע? ואַמא עַן מַוַאלִיד בֻּרג' אלאַסַד אֵיה! בֻּרגִ הַאת תַ )חַ תַ נשוּף עַלַא אלצעִיד אלמִהנִי לַא תַאח'ד' אַי קַרַארַ מַצִירִי פִי עַ מַ לַכ פַאִנתֵ לִסתַ מִסתעִדַ לִדַ 'לִכַ אלאַאן גַ עַמַל: יַלא יַא אַבּוּ סַנַד וַק אַנַא? אַנַא תַעבַּאן אליוֹם מֻש קַאדֵר אַח'ד' אַי קַרַאר מַצִירִי עַן אִדֵ 'נכםֹּ מַא רַח אַוקַע מֻש קַאדרֵה אַצַדק מֻש קַאדרֵה אַסתַועֵבּ אִנוֹ חִ 'סֵר אלבּוֹנֻס לִאַנוֹ מַא וַקעַ אלעַקד עַשַאן אִשִי סִמעוֹ בּאלאַבּרַאג' מִן חַלקת: סַאמַחתכ בַּאחִבּהַא בַּאחִבּהַא לַאִם סַנַד ואִלי בִּ יחִבּ בִּ יסַאמֵח אַבּוּ סַנַד אַנַא עַ 'לטַאנֵה בַּאערַף ואַנַא סוֹרי כתִיר כתִיר שוּ צַאיֵר חַבִּיבּתִי בּעַיוּנ מֻש מֻסתַאהלֵה לַא לַא לַא חַבִּיבִּי הַאדַא בַּצַל אלקַלאיֵה קַלאיֵה? הַאלמַרַה סַמַאח מִן חַלקת: שֻע'ל אלבֵּית בַּס לִלבַּנַאת טִלֵע אַכל אלבַּאבַּא אַזכַא מִן אַכל אלמַאמַא בַּס אלקַלאיֵה תַבּעֵת מַאמַא לַא יעלַא עַלֵיהַא בּצַרַאחַה כתִיר זַאכִי טַבִּיח' אלבַּאבַּא ואַנַא אִנצַדַמת כַמַאן בִּיטלַע מִנו צַדַק אַבּוּכי אַבּוּ סַנַד! סַנַד! בִּדכםֹּ מסַאעַדֵה? לַא לַא לַא אִרתַאחוּ אִנתוּ אִחנַא מדַבּרִין חַאלנַא מִין קַאל אִנוֹ שֻע'ל אלבֵּית בַּס לִלבַּנַאת? מִן חַלקת: סַ ע'ירַאן אַהלִי מַט'בּוּט מַרַאת בִּיעמלוּ דרַאמַא. מנִיח מַא חַדַא שַאפהםֹּ בִּהַאדַא אלמַנטַ בַּס בּצַרַאחַה בִּמַוצ'וּע אלסוֹשַאל מִידִיַא כַאן מַעהםֹּ חַק אַנַא בַּאערַף אִנוֹ אִלי עמִלתוֹ עַ 'לַט ומֻש רַח אַעִידהַא אִנפִגַ אִלי עִמֵל סַנַד חִ 'לַאל עִיד מִילַאדוֹ לַסַאמֵר סַנַד! בַּס אַול מַרַה בַּאערַף קַדיה )קַדיש( גַ אַהלִי בִּיחִבּוּנִי ואַנַא מֻש פַאהֵם לֵיש מַא בִּיחכוּ לִי אַכתַר מִן הַאדַא אלחַכִי
