 בַּס קוּלי לִי כִיף הֵיכ צַּאר מַּעֵכ? מֻש עַּארפֵה הַּאדַּאכ אליוֹם כַּאן יַּזַּן עִנדוֹ אמתִחַּאן קַּבלי בִאלט'בט בִאלט'בט ותאַח'ר בִתַּסלִימוֹ ובַּס עַּלַּא טוּל יַּעני אִנתי עמִלתי סַּבמִית? אַאה, בַּס כתִיר כֻנת מַּלח'וּמֵה טַּב אִנתי בַּעַּת'תי לִלמִס? סַּאַלתִיהַּא? אַאה אַאה בַּאעַּת'ת להַּא ובַּאסתַּנאהַּא תרֻד מַּעקוּל אַאח'ד' צִפר באמתִחַּאן נִהַּאיֵת אלסנֵה? חַּכַּת מַּעי אלמִס הַּלק ובִתקוּל אִנכ כתִיר מתאַח'רֵה בתַּסלים אלאמתִחַּאן ואצ'טַּרַּת מַּא תחֻט עַּלַּאמֵה עַּלַּא אלשהַּאדֵה כִיף יַּעני? כֻל הַּאדַּא לִאַני תאַח'רת? שוּ שוּ זַּנבי אַנַּא? יַּזַּן כַּאן עִנדוֹ אמתִחַּאן ומַּא עִנדנַּא עֵ כוֹמביוּתר וַּאחַּד יַּעני דִרַּאסֵת כֻל אלפַּצל רַּאחַּת עַּלַּא אלפַּאצ'י? ומעַּדלי, מעַּדלי שוּ רַּח פִיה? מַּא תִקלַּקי מַּאמַּא, אַנַּא רַּח אַחַּאוֵל אתוַּאצַּל מַּע אלמֻדירֵה ואַשוּף שוּ בנִקדַּר נִעמַּל רַּנַּא וַּלַּא יהִמכ אַּכיד פִיה חַּל סַּאלי שוּפתי טַּאבת אלקַּדַּם? סַּנַּד, מֻש שַּאיִפני מִנהַּמכֵה פִי אלכִתַּאבֵה? זַּחלֵק אעתַּדֵ עַּלַּא מקַּאטַּעת שֵיכסביר מַּא אַזנַּחַּ עַּם בַּאכתֵב אִשי מֻהִם מַּקַּאל יחַּאכי מַּשַּאכֵל אלעַּצר עַּם בַּאכתֵב עַּן אלדרַּאסֵה עַּן בֻעד אלדרַּאסֵה עַּן בֻעד? אִנתי מפַּכרֵה חַּאלֵכ? מעַּלמֵה וִלא חַּ 'בירֵה נַּפסִיה? סַּנַּד אִטלַּע וסַּכר אלבַּאב וַּרַּאכ טַּיב גַּ 'אובִיני, שוּפתי אלטאבֵה? מַּאמַּא שוּפי סַּנַּד טַּאלֵע טַּאלֵע אַינ כֻנא? נַּעַּם, מעַּאלַּגֵ תַּ 'ע'רַּ עַּן בֻעד סַּנַּד עַּם בִיזעַּג'ני ואַנַּא עַּם בַּאכתֵב מַּקַּאל מֻהִם כתִיר רַּח יסַּאעֵד רַּנַּא במֻשכִלִתהַּא אַחלַּא כַּאתבֵה אלעַּאלַּם אַנַּא רַּאיחַּה אַעמַּל לַּכ אִשי זַּאכי בַּס תחַּ אִבעַּתי אלמַּקַּאל עַּ שַּאן אַקרַּאוֹ קַּדיש בתִדפַּעי? שֻכרַּן סַּאלי בַּס בצַּרַּאחַּה מַּא אַטִ רַּח יעמַּל פַּרק רַּח תשוּפי, רַּח אַנַּזלוֹ עַּלַּא צַּפחֵת אלמַּדרַּסֵה ואַּכיד אלמֻדירֵה רַּח תִקרַּאוֹ ותתגַּ 'אוַּב מַּעוֹ כַּמַּאן סַּאלי אלכֻל בִיערַּף אִנכ צַּאחִבתי ומַּא רַּח יסמעוּ לֵכ אַצלַּן הַּלק ביטַּלעוּ לֵכ מִליון עַּ 'לטַּה מִליון? כֻל אלמַּקַּאל כִלמֵה חַּרַּאם עַּלֵיכי שֻכרַּן, בַּס עַּן גַּ מַּא תעַּ חַּאלֵכ רַּח אַּבעַּתוֹ אליוֹם וחַּ 'לינַּא נשוּף סַּאלי אלמַּקַּאל מֻש מַּעקוּל כתִיר מתַּ זֵ ופִיה נִקַּאט כתִיר חִלוֵה לתַּחסין עַּן בֻעד מַּאמַּא מַּא תעַּ חַּאלֵכ אלמַּקַּאל אנרַּפַּצ' עַּן גַּ לֵיש? מִין הַּאדַּא אִלי רַּפַּצ' לֵכ אלמַּקַּאל? בַּס אַעטִיני אִסמוֹ מַּא בַּאערַּף יִמכֵן זַּי מַּא קַּאל סַּנַּד אַנַּא מֻש מִח'תַּצה באלמַּוצ'וּע וזַּי מַּא קַּאלַּת רַּנַּא אלמַּקַּאל מַּא רַּח יעמַּל פַּרק אַצלַּן חַּ 'לַּצ בַּטלת בִדי אַּכתֵב לַּא יַּא בַּאבַּא, מֻש מח'תַּצה? הַּאדַּא אלמַּוצ'וּע בִיחֻ ונֻצ כַּטַּאלבֵה גַּ 'רַּבתי אלאִלכתרוֹני ורַּאיכֵ לַּאזֵם תִחכִיה מַּעוֹ חַּק אַּבוּכי שוּ רַּאיֵכ נטַּ עַּליה ונִבעַּתוֹ לִלגַּ 'רִידֵה סמִעת אִנוֹ עַּם בִישַּג'עוּ אלשבַּאב עַּלַּא כִתַּאבֵת מַּוַּאצ'יע מִתֵל הֵיכ, שוּ רַּאיֵכ? מַּאמַּא, אִזַּא אלמַּדרַּסֵה מַּא קִבלַּתוֹ, אלגַּ 'רִידֵה בִדהַּא תקַּבלוֹ יַּעני? סַּאלי אִנתי מַּוהוּבֵה בִאלכִתַּאבֵה וחַּאבה תסַּאעדי רַּנַּא, צַּח? מֻש בַּ רַּנַּא, אַנַּא חַּאבה אַסַּאעֵד כֻל אִלי עַּם ביֻדרסוּ עַּן בֻעד טַּיב חַּ 'לינַּא נגַּ 'רֵב, שוּ רַּח אוֹכֵי מוַּאפקַּה סַּאלי מַּאמַּא חַּתא לַּו רַּפַּצ'וּה אלגַּ 'רידֵה מֻש מֻהִם אִנתי מַּוהוּבֵה באלכִתַּאבֵה מַּא תתפַּשַּ אַּבדַּן ותזַּכרי דַּאימַּן אִנוֹ קִימתֵכ אַּכבַּר אלאַשיאַּא אִלי בתִעמלִיהַּא, אתפַּקנַּא? אתפַּקנַּא אַלו? מַּרחַּבַּא אַהלַּן, מַּעי? מֻמכֵן אַחכי מַּע סַּאלי? תפַּצ'ל אַנַּא סַּאלי סַּאלי אַנַּא אִסמי עַּבד אַללה מחַּרֵר אלגַּ 'רידֵה אִלי בַּעת'תי בִהַּא אלמַּקַּאל ואַנַּא קַּרַּאת אלמַּקַּאל בצַּרַּאחַּה חִלוּ גִ בַּס בִדוֹ תַּעדיל אוֹכֵי שֻכרַּן עַּמוֹ בַּס אִחנַּא שוּפנַּא קֻדֻרַּאת רַּאאִעַּה וחַּאבין תִתדַּרַּבי עִנדנַּא שַּהר באלגַּ 'רִידֵה ואַול אִשי רַּח תִעמלִיה אִנכ רַּח תטַּ אלמַּקַּאל מַּע אַחַּד אלזַּמִילַּאת אלמֻח'תַּצין עַּ שַּאן ננשרוֹ אלגַּ 'רידֵה, בִצַּרַּאחַּה אַפכַּארוֹ קַּ ימֵה וחֻ 'צוּצַּן אלמֻבַּאדַּרַּה אִלי אקתַּרַּחתִיהַּא עַּן תַּצליח אלכמביוּתֵרַּאת אלקַּדימֵה ותַּוזיעהַּא עַּלַּא עַּן גַּ שֻכרַּן שֻכרַּן יַּעני בַּאקדַּר אַדַּאוֵם בֻכרַּה? בֻכרַּה? שִכלֵכ מִסתַּעג'לֵה אַאה לַּא בַּס בִצַּרַּאחַּה חַּאבה אלמַּקַּאל בסֻרעַּה חַּ 'לַּצ בנִסתַּנאכי בֻכרַּה מֻמכֵן אַחכי מַּע אַהלֵכ ואַרַּתב אלתַּפַּאציל מַּעהם? מַּאמַּא! יַּא גַּ 'מַּאעַּה חַּאב אַחַּ אִנוֹ הַּלק אִגַּ אלמֻדיר אלעַּאם לִשִרכֵת אלכמביוּתר בִנַּאַן עַּלַּא מַּקַּאל סַּאלי קַּרַּר אלתבַּרע בִמִית גִ 'הַּאז כמביוּתר לִלמַּדַּארֵס עַּ שַּאן עַּן בֻעד אִזַּא עִנדכם חִלם אַו פִכרַּה תמסכוּ פִיהַּא ואִזַּא פשִלתוּ אַול מַּרַּה מַּא תִסתַּסלֵמוּ אסתַּמִרוּ ואשתִע'לוּ עַּלֵיהַּא טַּורוּהַּא לַּחַּד מַּא ואִנתוּ כַּאַהל לַּאזֵם תשַּג'עוּ ולַּאדכם דַּאימַּן אִנוֹ מַּא יִסתַּסלֵמוּ ולַּאקוּ טֻרֻק תסַּאעַּדוּהם פיהַּא חַּתא יחַּקקוּ אַחלַּאמהם ויטַּורוּ קֻדֻרַּאתהם
