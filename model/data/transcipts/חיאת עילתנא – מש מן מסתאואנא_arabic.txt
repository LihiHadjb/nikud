 הַאדַא הַאדַא מֻש כִיף יַעני מֻש אִחנַא מתפקִין אִנוֹ מִן הוֹן לַהוֹן הַאי אלחגַ בַס הַאי אִגַ גַ אלחַגַ יַעני באלעַארצַ הַאדַא מֻש מַחסוּב מַרחַבַא דִינַא יַעטִיכי אלעַאפיֵה, שִכלוֹ כַאן יוֹמֵכ מֻתעֵב כתִיר כַאן עִנדנַא צַ וֵין אלולַאד? רַאחוּ ילִעַבוּ באלחַארַה טַמננִי, צַאר מַע אלמֻדִירֵה אלג'דִידֵה? מֻש עַארֵף לֵיש? מַא להַא אִשי? מֻש קֻלת עִנדהַא חִ 'ברַ ומַעהַא דכֹתוֹרַאה ופַהמַאנֵה כתִיר מַט'בוּט בַס בַס אַנַא עַארֵ אלשבַאב פִי אלמַכתַב מִנזעגִ כתִיר ומֻש עַאגִ 'בהםֹ מַעהַא ואִנתֵ שוּ רַאיַכ? הֻם מֻש עַאגִ 'בהםֹ אִנוֹ מֻדִירִתהםֹ סִת בַס בִדכ אלצרַאחַה מִן יוֹם מַא אִגַ וַצ'ע אלשרכֵה כתִיר תחַסן ואלאֻמוּר תרַתבַת, בִדכ אלצרַאחַה? אַנַא שַאיֵף אִנהַא אַחסַן בכתִיר מִן אלמֻדיר אלקַדים בַס, אֵימתַא רַח תבַטל תֻחכֻם בִהַאי אלטרִיקַה? אַנַא הַאדי אַאחֵ מַרַה בַאלעַב מַעהםֹ שוּ צַאיֵר יַא ולַאד? בַעד מַא דַח'לנַא אַמא שוּ בַאבַא קַאל מַא חַסַבוּה וח'סִרנַא מִין הֻם? ולַאד אלחַארַה אִלי פוֹק ולֵיש אִנתוּ אַצלַן רוּחתוּ תִלעַבוּ מַעהםֹ? בַאבַא, עַמוֹ אַבוּ יַזַן מֻש עַאג'בוֹ אִנהםֹ לִעבוּ מַעהםֹ בִיחכִי אִנוֹ סֻכאן אלחַארַה אלפוֹק מֻש זַינַא בַס הֻם שַאטרִין כתִיר לִעבהםֹ חִלוּ ובַעדֵין מַע הַאלחַכי, שוּ יַעני זַינַא ומֻש זַינַא? מִן אֵ ימתַא בִנמַיז בֵין בחַסַב וֵין סַאכנִין? יַללה אֻדח'לוּ עַ 'סלוּ אִידֵיכםֹ, אַנַא ואלבַאבַא רַח נחַצ'ר אלאַכל אַנַא בַאקוּל אלחַל אִנוֹ נִעמַל מֻבַאדַרַה ונטַ אלמַלעַב ונִשתרי מַרמַא ג'דִיד אַיוַא, מֻשכִלִתנַא אִנוֹ מֻש קַאדרִין נחַדד צַח עַ שַאן הֵיכ בִנצַ נִתחַ 'אנַק בַס אַהַאלִינַא, אִזַא שַאפוּנַא עַם נִעמַל אִשי מַע בַעצ' רַח ינזִ בַעדין מִן וֵין אַצלַן רַח מַצַארי נִשתרי מֻש מֻשכִלֵה, שוּ רַאיכם נִעמַל בַאזַאר פִיה כֻל אלאַשיאַ אִלי מַא בנִחתַאג'הַא? ומֻמכֵן נִעמַל כֵיכ ועַציר אַנַא, אֻח'תי בתִעמַל כֵיכ זַאכי ואִחנַא אִזַא בנִתסַאעַד, רַח נטַ אלמַלעַב ורַח נִנבסֵט כֻל ורַח נעמַל דַורִי כֻל יוֹם ורַח נצַ נעַ יַללה מַא אַחלַאהם אלולַאד בִיסַלוּ חַאלהםֹ. בַס מִין הַאדוֹל אלולַאד אִלי מַעַאהםֹ? יִסלַמוּ אִידֵיכי עַלַא האלקַהוֵה אלזאכיֵה, מַא אַחלַא רִיחִתהַא צַחתֵין והַנַא בתִערפי יַא אִם יַזן, שִכלוֹ הַאדוֹל אלולַאד מִן אלחַארַה אלפוֹקנִיה מַא בִדכםֹ תִערפוּ סמִעת, בִיקולו אִנוֹ פִיה עֵילֵה מִן אלחַארַה אלפוֹקַנִיה, בִדהםֹ יֻסכֻנוּ אלפַאצ'יֵה אִלי באלעַמַארַה מַא לֵכ נַקזתי? יצִירוּ גִ 'ירַאננַא יַעני? בַס מַא הֻם בִאלמַרַה מֻש זַינַא לֵיש לַאזֵם יכוּנוּ זַינַא יַא אִם יַזַן? לַא, יַעני יִמכֵן יכוּנוּ מנַאח, בַס עַאדי יַעני ולַאדהם יִלעַבוּ מַע ולַאדכם? שוּ פִיהַא יַעני? אִשרַבוּ אלקַהוַה, בַלַאש תֻברדֹ רַאיחַה אַג'יב הַרִיסה בִתשַהי, בַס בלִיז, אִזַא טִלעוּ אלולַאד, נַאדוּני בסֻרעַה חַ אַסתַג'וִבהםֹ חַאצֵ יַא בֻרג' אלמֻרַאקַבֵה אִם אִנצַאף אִג'ו? לַא בַס בִדי אַקוּל לֵכ, אִנוֹ מַא בִדנַא נעַ 'לבֵכ וַלַא עַ 'לַבֵה וַלַא אִשי תפַצ'לוּ יַא עַלי, מִן אללח'מֵה גִ אלשוֹכֵה אלעַ 'לַט לֵיש מַא חַ 'לַצ, אכלֹ בִאַי אִשי לַא, יַעני, בַס הֻם הַאדוֹל אִלי בַאחַ לִלבִנת אִלי בתִיג'י בִתנַט'ף לִי אלבֵית מַא תַאַאחִ מַא אִנתוּ לַח'מתוּני לַמא נַאדַתוּני הֵיכ יַעני מֻש פַאהמֵה, אִנתי מַא בתִ ג'לִיהם?
